```sql

CREATE TABLE users (
  id UUID PRIMARY KEY,
  email TEXT UNIQUE,
  subscription_tier TEXT,
  stripe_customer_id TEXT
);

-- 海报表
CREATE TABLE posters (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users,
  title TEXT,
  products JSONB,
  template TEXT,
  created_at TIMESTAMP
);

-- 订阅表
CREATE TABLE subscriptions (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users,
  stripe_subscription_id TEXT,
  status TEXT,
  current_period_end TIMESTAMP
);
```
